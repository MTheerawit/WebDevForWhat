create table COMPANY (
    symbol varchar(10) primary key,
    set50 varchar(4),

create table TRADE (
    date date,
    symbol varchar(10),
    open float(24),
    high float(24),
    low float(24),
    close float(24),
    percentChange float(24),
    volume float(53),
    money float(53),
    macd float(24),
    rsi float(24),
    status varchar(4),
    primary key(date, symbol),
    foreign key (symbol) references COMPANY(symbol)
);
